package pruebaQA;

import io.restassured.http.ContentType;
import org.junit.Test;

import static io.restassured.RestAssured.given;
import static org.hamcrest.Matchers.*;


public class AutomationTest {



    @Test
    public void requestDecision_checkStatusCode () {

        given().
                when().
                get("https://yesno.wtf").
                then().
                assertThat().
                statusCode(200);
    }

    @Test
    public void requestDecision_checkContentType() {

        given().
                when().
                get("https://yesno.wtf").
                then().
                assertThat().
                contentType(ContentType.HTML);

    }

    @Test
    public void requestDecision_logRequest() {

        given().
                log().all().
                when().
                get("https://yesno.wtf").
                then().
                log().body();

    }

    @Test
    public void requestDecision_checkAnswerInResponseBody_ExpectYes() {

        given().
                when().
                get("https://yesno.wtf").
                then().
                assertThat().
                body("@code", equalTo("yes"));

    }

    @Test
    public void requestDecision_checkAnswerInResponseBody_ExpectNo() {

        given().
                when().
                get("https://yesno.wtf").
                then().
                assertThat().
                body("code", equalTo("no"));

    }

    @Test
    public void requestDecision_checkAnswerInResponseBody_ExpectMaybe() {

        given().
                when().
                get("https://yesno.wtf").
                then().
                assertThat().
                body("code", equalTo("maybe"));

    }
}
